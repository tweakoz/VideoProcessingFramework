cmake_minimum_required(VERSION 3.10)

project(Video_Processing_Framework)

set(TRACK_TOKEN_ALLOCATIONS FALSE CACHE BOOL "Debug memory allocations within VPF")

if(TRACK_TOKEN_ALLOCATIONS)
	add_definitions(-DTRACK_TOKEN_ALLOCATIONS)
endif(TRACK_TOKEN_ALLOCATIONS)

#Undef MIN & MAX & C runtime warnings for Windows
if(WIN32)
	add_definitions(-DNOMINMAX)
	add_definitions(-D_CRT_SECURE_NO_WARNINGS)
endif(WIN32)

if(UNIX)
	set (CMAKE_CXX_STANDARD 11)
	include_directories(/usr/local/cuda/include)
	link_directories(/usr/local/cuda/lib64)
endif(UNIX)

add_subdirectory(PyNvCodec)

#Add C++ sample
include_directories(${TC_CORE_INC_PATH})
include_directories(${TC_INC_PATH})
include_directories(${VIDEO_CODEC_SDK_INCLUDE_DIR})

#Bring all the stuff together
install(FILES $<TARGET_FILE:TC_CORE>									DESTINATION lib)
install(FILES $<TARGET_FILE:TC>											DESTINATION lib)

if(GENERATE_PYTHON_BINDINGS)
	install(FILES $<TARGET_FILE:PyNvCodec>								DESTINATION lib)
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleDecode.py			DESTINATION share/vpf )
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleEncode.py			DESTINATION share/vpf )
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleTranscode.py		DESTINATION share/vpf )
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleFrameUpload.py		DESTINATION share/vpf )
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleColorConversion.py	DESTINATION share/vpf )
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleSufraceDownload.py	DESTINATION share/vpf )
	install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/SampleTranscodeOneToN.py	DESTINATION share/vpf )
endif(GENERATE_PYTHON_BINDINGS)
